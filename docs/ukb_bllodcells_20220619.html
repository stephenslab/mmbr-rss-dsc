## pairwise sharing
get_pairwise_sharing = function(dat){
  R = ncol(dat)
  S = diag(R)
  for(i in 1:(R-1)){
    for(j in (i+1):R){
      sig_i=which(dat[,i] == 1)
      sig_j=which(dat[,j] == 1)
      S[i,j] = length(intersect(sig_i, sig_j))/length(union(sig_i,sig_j))
    }
  }
  S[lower.tri(S, diag = FALSE)] = t(S)[lower.tri(S, diag = FALSE)]
  colnames(S) = row.names(S) = colnames(dat)
  return(S)
}
